server.js

import express from "express";
import path from "path";
import { fileURLToPath } from "url";

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);
const app = express();
const PORT = process.env.PORT || 5000;

app.use(express.static(path.join(__dirname, "public")));

const send = (res, file) =>
  res.sendFile(path.join(__dirname, "public", file));

app.get("/",      (_, res) => send(res, "index.html"));
app.get("/demo",  (_, res) => send(res, "demo.html"));
app.get("/docs",  (_, res) => send(res, "docs.html"));
app.get("/sign-in", (_, res) => send(res, "sign-in.html"));

// Simple 404 → homepage (so non-existent routes still render shell)
app.use((_, res) => send(res, "index.html"));

app.listen(PORT, () => {
  console.log(`Skyla site running on http://localhost:${PORT}`);
});
